
        int main()
        {
            string str;
            int i, frequency; frequency = 0;
            printf("Enter a string: ");
            gets(str);
            for(i = 0; str[i] != ""; i = i + 1)
            {
                if("a" == str[i])
                    frequency = frequency + 1;
            }
            printf("Frequency of %c = %d", "a", frequency);
            return 0;
        }
