
        int main()
        {
            int i, n;
            float arr[100];
            printf("Enter total number of elements(1 to 100): ");
            n = 100;
            printf("\n");
            // Stores number entered by the user
            for(i = 0; i < n; i = i + 1)
            {
                arr[i] = rand() ;
            }
            // Loop to store largest number to arr[0]
            for(i = 1; i < n; i = i + 1)
            {
                // Change < to > if you want to find the smallest element
                if(arr[0] < arr[i])
                    arr[0] = arr[i];
            }
            printf("Largest element = %.2f", arr[0]);
            return 0;
        }
