
            // Check prime number
            boolean checkPrime(int n) {
                int i;
                boolean isPrime; 
                isPrime = true;
                for(i = 2; i <= n/2; i = i + 1) {
                    if(n % i == 0) {
                        isPrime = false;
                        break;
                    }
                }
                return isPrime;
            }
