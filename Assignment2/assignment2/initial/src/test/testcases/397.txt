
            int main() {
                int n, t1, t2, nextTerm;
                for (i = 1; i <= n; i = i + 1) {
                     // Prints the first two terms.
                    if ((i == 1) || (i == 2)) {
                        print( "%d",t1);
                        continue;
                    }
                    nextTerm = t1 + t2;
                    t1 = t2;
                    t2 = nextTerm;
                    print("%d ",nextTerm);
                }
                return 0;
            }
