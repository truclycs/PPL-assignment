
        boolean notEqual(boolean a, boolean b){
            return a != b;
        }

        boolean notEqualInt(int a, int b, int c){
            return a != c || b != c;
        }

        boolean main(boolean a, boolean b){
            return notEqual(a, b);
        }

        boolean notEqualFloat(int d, int e, float f, float g){
            notEqualInt(d, e, e);
            notEqualFloat(d, e, f, g);
            return f != g;
        }
