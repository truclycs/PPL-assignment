 
        void main() {
            int i, n, res;
            res = 1;
            for (i = 0; i < n; i = i + 1) {
                if (n % i == 0) {
                    res = res * i;
                    continue;
                }
                else {
                    if (n % i == 1) {
                        break;
                    }
                    else {
                        res = res + i;
                    }
                }
            }
        }
