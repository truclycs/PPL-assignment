 
        boolean checkPrime(int n) {
            if (n < 2) {
                return false;
            } 
            int i;
            for (i = 2; i * i <= n ; i = i + 1) {
                if (n % i == 0) {
                    return 0;
                }
            }
            return true;
        }
        void main() {
            int n;
            checkPrime(n);
        }
