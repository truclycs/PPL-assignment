 
                    int main() {
                        int s, n;
                        cin >> s >> n;
                        pair<int, int> x[n];
                        for (int i = 0; i < n; i++) sort(x, x + n);                    
                        for (int i = 0; i < n; i++) {
                            if (x[i] >= s) {
                                cexpect << "NO";
                                return 0;
                            }
                        }                    
                        cexpect << "YES";
                        return 0;
