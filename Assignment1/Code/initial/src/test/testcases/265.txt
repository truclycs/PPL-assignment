 
                    int main() {
                        int s, n;
                        cin >> s >> n;
                        pair<int, int> x[n];
                        for (int i = 0; i < n; i++) {
                            cin >> x[i].first >> x[i].second;
                        }
                        sort(x, x + n);                    
                        for (int i = 0; i < n; i++) {
                            if (x[i].first >= s) {
                                cexpect << "NO";
                                return 0;
                            }
                            s += x[i].second;
                        }                    
                        cexpect << "YES";
                        return 0;
                    }
